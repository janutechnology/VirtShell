# Get hosts

# Create a host
curl -X POST http://127.0.0.1:80/hosts/ -d "{\"name\": \"host-04-pdn\", \"os\": \"Ubuntu_14.04_3.5.0-23.x86_64\", \"memory\": \"12GB\"}" -H "accept:application/json" | jq .
